24inline double ProbabilityToRate(double probability, int time = 1) {
25 if (probability < 0 || probability >= 1) {
26 throw std::domain_error("Out of probability value range");
27 }
28 return -log(1 - probability) / time;
29}

34inline double RateToProbability(double rate, int time = 1) {
35 if (rate < 0) {
36 throw std::domain_error("Out of rate value range");
37 }
38 return 1 - exp(-rate * time);
39}

46inline double Discount(double value, double discount_rate, double timestep,
47 bool annual = false) {
48 if (discount_rate < 0 || timestep < 0) {
49 throw std::domain_error("Out of discount rate or Timestep Range");
50 }
51 discount_rate = annual ? discount_rate / 12 : discount_rate;
52 double denominator = std::pow(1 + discount_rate, timestep);
53 return value / denominator;
54}
