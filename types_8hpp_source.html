22inline static const std::string
23POPULATION_HEADERS(bool pregnancy = false, bool hcc = false,
24 bool overdose = false, bool hiv = false, bool moud = false) {
25 std::stringstream headers;
26 // basic characteristics
27 headers << "sex,age,is_alive,boomer_classification,death_reason,";
28 // BehaviorDetails
29 headers << "drug_behavior,time_last_active_drug_use,";
30 // HCVDetails
31 headers << "hcv,fibrosis_state,is_genotype_three,seropositive,time_hcv_"
32 "changed,time_fibrosis_state_changed,times_hcv_infected,times_"
33 "acute_cleared,svrs,";
34 // HIVDetails
35 std::string hiv_details =
36 hiv ? "hiv,time_hiv_changed,low_cd4_months_count," : "none,-1,0,";
37 headers << hiv_details;
38 // HCCDetails
39 std::string hcc_details = hcc ? "hcc_state,hcc_diagnosed," : "none,false,";
40 headers << hcc_details;
41 // overdose characteristics
42 std::string overdoses =
43 overdose ? "currently_overdosing,num_overdoses," : "false,0,";
44 headers << overdoses;
45 // MOUDDetails
46 std::string moud_details =
47 moud ? "moud_state,time_started_moud,current_moud_state_concurrent_"
48 "months,total_moud_months,"
49 : "none,-1,0,0,";
50 headers << moud_details;
51 // PregnancyDetails
52 std::string pregnancy_details =
53 pregnancy ? "pregnancy_state,time_of_pregnancy_change,pregnancy_count,"
54 "num_infants,num_stillbirths,num_infant_hcv_exposures,num_"
55 "infant_hcv_infections,num_infant_hcv_tests,"
56 : "na,-1,0,0,0,0,0,0,";
57 headers << pregnancy_details;
58 // StagingDetails
59 headers << "measured_fibrosis_state,had_second_test,time_of_last_staging,";
60 // LinkageDetails
61 headers << "hcv_link_state,time_of_hcv_link_change,hcv_link_count,";
62 if (hiv) {
63 headers << "hiv_link_state,time_of_hiv_link_change,hiv_link_count,";
64 } else {
65 headers << "never,-1,0,";
66 }
67 // ScreeningDetails
68 headers << "time_of_last_hcv_screening,num_hcv_ab_tests,num_hcv_rna_tests,"
69 "hcv_antibody_positive,hcv_identified,time_hcv_identified,num_"
70 "hcv_identifications,hcv_screening_type,";
71 if (hiv) {
72 headers
73 << "time_of_last_hiv_screening,num_hiv_ab_tests,num_hiv_rna_tests,"
74 "hiv_antibody_positive,hiv_identified,time_hiv_identified,num_"
75 "hiv_identified,hiv_screening_type,";
76 } else {
77 headers << "-1,0,0,false,false,-1,0,na";
78 }
79 // TreatmentDetails
80 headers << "initiated_hcv_treatment,time_of_hcv_treatment_initiation,num_"
81 "hcv_treatment_starts,num_hcv_treatment_withdrawals,num_hcv_"
82 "treatment_toxic_reactions,num_completed_hcv_treatments,num_hcv_"
83 "salvages,in_hcv_salvage_treatment,";
84 if (hiv) {
85 headers << "initiated_hiv_treatment,time_of_hiv_treatment_initiation,"
86 "num_hiv_treatment_starts,num_hiv_treatment_withdrawals,num_"
87 "hiv_treatment_toxic_reactions,";
88 } else {
89 headers << "false,-1,0,0,0,";
90 }
91 // Utility - NOTE: While LifetimeUtility values are listed here, they cannot
92 // be assigned when creating a new Person. They are effectively read-only.
93 headers << "behavior_utility,liver_utility,treatment_utility,background_"
94 "utility,";
95 if (hiv) {
96 headers << "hiv_utility,";
97 } else {
98 headers << "1.0,";
99 }
100 headers << "min_utility,mult_utility,discounted_min_utility,discounted_"
101 "mult_utility,";
102 // Lifespan
103 headers << "life_span,discounted_life_span";
104 return headers.str();
105}
