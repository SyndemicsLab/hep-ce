43 void SelectFromDatabase(
44 const std::string &query,
45 std::function<void(std::any &storage, const SQLite::Statement &stmt)>
46 callback,
47 std::any &storage,
48 const std::unordered_map<int, std::variant<int, double, std::string>>
49 &bindings) const {
50 try {
51 SQLite::Database db(_database_file.string(),
52 SQLite::OPEN_READWRITE | SQLite::OPEN_CREATE);
53 SQLite::Statement stmt(db, query);
54
55 for (const auto &[index, value] : bindings) {
56 if (value.index() == 0) {
57 stmt.bind(index, std::get<int>(value));
58 } else if (value.index() == 1) {
59 stmt.bind(index, std::get<double>(value));
60 } else {
61 stmt.bind(index, std::get<std::string>(value));
62 }
63 }
64
65 SQLite::Transaction transaction(db);
66
67 while (stmt.executeStep()) {
68 callback(storage, stmt);
69 }
70
71 transaction.commit();
72 } catch (const std::exception &e) {
73 throw std::runtime_error("Error executing query: " + query + "\n" +
74 e.what());
75 }
76 }
