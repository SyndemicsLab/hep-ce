100inline bool FindInEventList(std::string event_name,
101 datamanagement::ModelData &model_data) {
102 auto event_list = utils::ToLowerVector(utils::SplitToVecT<std::string>(
103 model_data.GetFromConfig("simulation.events"), ','));
104 auto event_name_lower = utils::ToLower(event_name);
105 if (std::find(event_list.begin(), event_list.end(), event_name_lower) !=
106 event_list.end()) {
107 return true;
108 }
109 return false;
110}
